<script>
// ── CONFIG ────────────────────────────────────────────────────
const REFRESH_INTERVAL = 60; // seconds
let refreshTimer = REFRESH_INTERVAL;
let refreshInterval = null;
let countdownInterval = null;

// ── CLOCK ─────────────────────────────────────────────────────
function updateClock() {
  const now = new Date();
  document.getElementById('clock').textContent =
    now.toLocaleTimeString('en-US', { hour12: false,
      hour: '2-digit', minute: '2-digit', second: '2-digit' });
}
setInterval(updateClock, 1000);
updateClock();

// ── STATUS ────────────────────────────────────────────────────
function setStatus(state) {
  const dot  = document.getElementById('statusDot');
  const text = document.getElementById('statusText');
  dot.className = 'status-dot ' + (state === 'ok' ? '' : state);
  text.textContent = state === 'ok' ? 'LIVE' : state === 'loading' ? 'LOADING' : 'ERROR';
  text.style.color = state === 'error' ? 'var(--red)' : '';
}

// ── REFRESH BAR ───────────────────────────────────────────────
function startRefreshCountdown() {
  refreshTimer = REFRESH_INTERVAL;
  const bar = document.getElementById('refreshBar');
  const label = document.getElementById('refreshLabel');
  if (countdownInterval) clearInterval(countdownInterval);
  countdownInterval = setInterval(() => {
    refreshTimer = Math.max(0, refreshTimer - 1);
    const pct = (refreshTimer / REFRESH_INTERVAL) * 100;
    bar.style.width = pct + '%';
    label.textContent = 'REFRESH ' + refreshTimer + 's';
  }, 1000);
}

// ── UTILS ─────────────────────────────────────────────────────
function colorClass(val, positiveIsGood = true) {
  if (val === null || val === undefined || isNaN(val)) return 'c-muted';
  if (positiveIsGood) return val > 0 ? 'c-green' : val < 0 ? 'c-red' : 'c-muted';
  return val > 0 ? 'c-red' : val < 0 ? 'c-green' : 'c-muted';
}

function arrow(val) {
  if (val === null || val === undefined || isNaN(val)) return '';
  return val > 0 ? ' ▲' : val < 0 ? ' ▼' : ' –';
}

function fmt(val, decimals = 2, suffix = '') {
  if (val === null || val === undefined) return '--';
  return Number(val).toFixed(decimals) + suffix;
}

function fmtTime(iso) {
  if (!iso) return '--';
  try {
    return new Date(iso).toLocaleTimeString('en-US',
      { hour: '2-digit', minute: '2-digit', hour12: false });
  } catch { return '--'; }
}

function fmtRelTime(iso) {
  if (!iso) return '--';
  try {
    const diff = Math.floor((Date.now() - new Date(iso)) / 60000);
    if (diff < 1)  return 'JUST NOW';
    if (diff < 60) return diff + 'MIN AGO';
    const h = Math.floor(diff / 60);
    return h + 'HR AGO';
  } catch { return '--'; }
}

// ── FETCH WRAPPER ─────────────────────────────────────────────
async function apiFetch(endpoint) {
  const res = await fetch(endpoint, { cache: 'no-store' });
  if (!res.ok) throw new Error(`HTTP ${res.status}`);
  return res.json();
}

// ── RENDER: REGIME ────────────────────────────────────────────
function renderRegime(data) {
  const label       = data.label || 'UNKNOWN';
  const confidence  = data.confidence_score ?? null;
  const indicators  = data.indicator_breakdown || [];
  const risks       = data.key_risks || [];
  const positioning = data.asset_class_positioning || [];
  const ts          = data.timestamp || null;

  document.getElementById('regimeTimestamp').textContent =
    ts ? 'AS OF ' + fmtTime(ts) : '--';

  // Build indicator cards
  const indCards = indicators.map(ind => {
    const cls = ind.signal === 'BULLISH' ? 'c-green'
              : ind.signal === 'BEARISH' ? 'c-red' : 'c-amber';
    return `
      <div class="regime-ind">
        <div class="regime-ind-name">${ind.name || ''}</div>
        <div class="regime-ind-val ${cls}">${ind.value !== undefined ? ind.value : '--'}</div>
        <div class="regime-ind-signal ${cls}">${ind.signal || ''}</div>
      </div>`;
  }).join('');

  // Build risks
  const riskItems = risks.length
    ? risks.map(r => `<div class="regime-risk-item">▸ ${r}</div>`).join('')
    : '<div class="c-muted" style="font-size:10px">NO FLAGS</div>';

  // Build positioning
  const posItems = positioning.map(p => {
    const cls = p.stance === 'OW' ? 'OW' : p.stance === 'UW' ? 'UW' : 'N';
    const prefix = p.stance === 'OW' ? '▲ ' : p.stance === 'UW' ? '▼ ' : '– ';
    return `<div class="pos-item ${cls}">${prefix}${p.asset_class || ''}</div>`;
  }).join('');

  document.getElementById('regimeContent').innerHTML = `
    <div class="regime-block">
      <div class="regime-label-wrap">
        <div class="regime-tag ${label}">${label}</div>
        <div class="regime-confidence">
          CONFIDENCE <span>${confidence !== null ? fmt(confidence, 0) + '%' : '--'}</span>
        </div>
      </div>
      <div>
        <div class="regime-indicators">${indCards || '<div class="c-muted">No indicator data</div>'}</div>
        ${posItems ? `<div class="regime-positioning"><div class="regime-pos-title">SUGGESTED POSITIONING</div><div class="pos-grid">${posItems}</div></div>` : ''}
      </div>
      <div class="regime-risks">
        <div class="regime-risks-title">KEY RISKS</div>
        ${riskItems}
      </div>
    </div>`;
}

// ── RENDER: MACRO ─────────────────────────────────────────────
function renderMacro(data) {
  const series = data.series || data;
  const ts = data.timestamp || null;
  document.getElementById('macroTimestamp').textContent =
    ts ? 'AS OF ' + fmtTime(ts) : '--';

  // Expect array or object of series
  const items = Array.isArray(series) ? series : Object.values(series);
  if (!items.length) {
    document.getElementById('macroContent').innerHTML =
      '<div class="error-row">NO MACRO DATA</div>';
    return;
  }

  const cells = items.map(s => {
    const chg = s.change ?? null;
    const positiveGood = s.positive_is_good !== false;
    const cls = colorClass(chg, positiveGood);
    return `
      <div class="macro-cell">
        <div class="macro-cell-label">${s.label || s.id || ''}</div>
        <div class="macro-cell-value ${cls}">${s.current !== undefined && s.current !== null ? fmt(s.current, s.decimals ?? 2) + (s.suffix || '') : '--'}</div>
        <div class="macro-cell-sub">${s.description || ''}</div>
        ${chg !== null ? `<div class="macro-cell-change ${cls}">${chg >= 0 ? '▲' : '▼'} ${fmt(Math.abs(chg), s.decimals ?? 2)}${s.suffix || ''} vs PRIOR</div>` : ''}
      </div>`;
  }).join('');

  document.getElementById('macroContent').innerHTML = cells;
}

// ── RENDER: YIELDS ────────────────────────────────────────────
function renderYields(data) {
  const yields   = data.yields || [];
  const spreads  = data.spreads || [];
  const ts       = data.timestamp || null;
  document.getElementById('yieldsTimestamp').textContent =
    ts ? 'AS OF ' + fmtTime(ts) : '--';

  // Bar chart
  const maxVal = Math.max(...yields.map(y => Math.abs(y.value || 0)), 0.01);
  const barHTML = yields.map(y => {
    const heightPct = Math.max(5, Math.round((Math.abs(y.value || 0) / maxVal) * 100));
    const inverted  = y.value < 0;
    return `
      <div class="yield-bar-wrap">
        <div class="yield-bar-val ${inverted ? 'c-red' : 'c-green'}">${fmt(y.value, 2)}%</div>
        <div class="yield-bar ${inverted ? 'inverted' : ''}" style="height:${heightPct}%"></div>
        <div class="yield-bar-label">${y.label || y.maturity || ''}</div>
      </div>`;
  }).join('');
  document.getElementById('yieldBars').innerHTML = barHTML ||
    '<div class="c-muted" style="font-size:10px;padding:20px">NO YIELD DATA</div>';

  // Spread table
  const spreadHTML = spreads.map(sp => {
    const cls = colorClass(sp.value, false); // inversion is bearish
    return `
      <div class="yield-spread-item">
        <span class="yield-spread-label">${sp.label || ''}</span>
        <span class="yield-spread-val ${cls}">${fmt(sp.value, 2)}%</span>
      </div>`;
  }).join('');
  document.getElementById('yieldSpreads').innerHTML = spreadHTML ||
    '<div class="c-muted" style="font-size:10px">NO SPREAD DATA</div>';
}

// ── RENDER: NEWS ──────────────────────────────────────────────
function renderNews(data) {
  const articles = data.articles || data;
  const ts = data.timestamp || null;
  document.getElementById('newsTimestamp').textContent =
    ts ? 'UPDATED ' + fmtRelTime(ts) : '--';

  if (!articles.length) {
    document.getElementById('newsContent').innerHTML =
      '<div class="error-row">NO NEWS DATA</div>';
    return;
  }

  const items = articles.map(a => `
    <a class="news-item" href="${a.url || '#'}" target="_blank" rel="noopener">
      <div class="news-time">${fmtRelTime(a.publishedAt || a.timestamp)}</div>
      <div class="news-headline">${a.title || a.headline || ''}</div>
      <div class="news-source">${(a.source?.name || a.source || '').toUpperCase()}</div>
    </a>`).join('');

  document.getElementById('newsContent').innerHTML = items;
}

// ── RENDER: ERROR ─────────────────────────────────────────────
function renderError(id, msg) {
  const el = document.getElementById(id);
  if (el) el.innerHTML = `<div class="error-row">⚠ ${msg}</div>`;
}

// ── FETCH ALL ─────────────────────────────────────────────────
async function fetchAll() {
  setStatus('loading');
  let anyError = false;

  const jobs = [
    apiFetch('/api/regime')
      .then(renderRegime)
      .catch(e => { renderError('regimeContent', 'REGIME UNAVAILABLE — ' + e.message); anyError = true; }),

    apiFetch('/api/macro')
      .then(renderMacro)
      .catch(e => { renderError('macroContent', 'MACRO DATA UNAVAILABLE — ' + e.message); anyError = true; }),

    apiFetch('/api/yields')
      .then(renderYields)
      .catch(e => { renderError('yieldsContent', 'YIELD DATA UNAVAILABLE — ' + e.message); anyError = true; }),

    apiFetch('/api/news')
      .then(renderNews)
      .catch(e => { renderError('newsContent', 'NEWS UNAVAILABLE — ' + e.message); anyError = true; }),
  ];

  await Promise.allSettled(jobs);
  setStatus(anyError ? 'error' : 'ok');
  startRefreshCountdown();
}

// ── AUTO REFRESH ──────────────────────────────────────────────
function startAutoRefresh() {
  if (refreshInterval) clearInterval(refreshInterval);
  refreshInterval = setInterval(fetchAll, REFRESH_INTERVAL * 1000);
}

// ── INIT ──────────────────────────────────────────────────────
fetchAll();
startAutoRefresh();
